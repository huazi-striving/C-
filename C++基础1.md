# 1.输入输出换行

~~~c++
#include<iostream>
using namespace std;  //必写
int main()
{
	cout << "hello world" << endl;  //自带换行
	cout << "hello world\n";  //\n换行
	cout << "hello world";   //不换行

	cout << "\n";   //换行
	cout << endl;   //换行


	int a;
	cout << "请输入一个数字：";
	cin >> a;     //输入
	cout << "number="<< a << endl;
	return 0;
}
~~~

# 2.变量,常量

~~~c++
#include<iostream>
using namespace std;  //必写
#define Day 7 //宏定义常量
int main()
{
	cout << Day << endl;

	int num1 = 100;   //定义变量
	cout << "num1=" << num1 << endl;
	num1 = 101;     //修改变量值，可以修改
	cout << "num1=" << num1 << endl;

	const int num2 = 10;  //const定义常量
	cout << "num2=" << num2 << "\n";
	//num2 = 11;      const定义的常量不能修改
	cout << "num2=" << num2 << endl;
	return 0;
}
~~~

# 4.sizeof()

~~~c++
#include<iostream>
using namespace std;  //必写
#define Day 7 //宏定义常量
int main()
{
	short num1 = 1;
	int num2 = 2;
	long num3 = 3;
	long long num4 = 4;

	cout << sizeof(num1) << endl;  //2字节
	cout << sizeof(num2) << endl;  //4字节
	cout << sizeof(num3) << endl;  //4字节
	cout << sizeof(num4) << endl;  //8字节
	cout << endl;
	cout << sizeof(short) << endl;  //2字节
	cout << sizeof(int) << endl;    //4字节
	cout << sizeof(long) << endl;   //4字节
	cout << sizeof(long long) << endl;  //8字节
	return 0;
}
~~~

# 6.强转

- 强转,a字符的ascall码的整数型

~~~c++
#include<iostream>
using namespace std;  //必写
int main()
{
	char str1 = 'a';
	cout << str1 << endl;   //a
	cout << (int)str1 << endl;  //97
	cout << sizeof(char) << endl;  //1字节
	return 0;
}
~~~

# 8.bool型

- 只要非0都代表真

~~~c++
#include<iostream>
using namespace std;  //必写
int main()
{
	bool flag = true;
	cout << flag << endl;  //1

	flag = false;
	cout << flag << endl;  //0
	return 0;
}
~~~
